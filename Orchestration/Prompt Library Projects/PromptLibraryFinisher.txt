PowerShell module
#requires -Version 5.1
Set-StrictMode -Version Latest
using namespace System.IO

# Module goals:
# - Load/validate YAML prompts, render templates, index/search, and export to CSV/Excel.
# - PS 5.1 + 7+ compatible. No external deps required (YAML via .NET + fallback).
# - Inline comments for clarity. Avoid $matches.Value gotcha and $var: scope colon parse issues.

# Region: Module Paths
$Script:RepoRoot   = Split-Path -Parent $PSScriptRoot
$Script:PromptRoot = Join-Path $RepoRoot 'prompts'
$Script:SchemaRoot = Join-Path $Script:PromptRoot 'schemas'
$Script:IndexRoot  = Join-Path $Script:PromptRoot 'index'

# Ensure folders exist
foreach($p in @($Script:PromptRoot,$Script:SchemaRoot,$Script:IndexRoot)){ if(-not (Test-Path $p)){ New-Item -ItemType Directory -Force -Path $p | Out-Null } }

function Resolve-Yaml {
    <#
    .SYNOPSIS
      Parse YAML into a hashtable using the best available loader.
    .NOTES
      Prefers PowerShell 7's ConvertFrom-Yaml; otherwise tries YamlDotNet if present; otherwise fails cleanly.
    #>
    [CmdletBinding()] param(
        [Parameter(Mandatory)][string]$Path
    )
    if (-not (Test-Path -LiteralPath $Path)) { throw "YAML not found: $($Path)" }
    $raw = Get-Content -LiteralPath $Path -Raw -Encoding UTF8

    if (Get-Command ConvertFrom-Yaml -ErrorAction SilentlyContinue) {
        return ConvertFrom-Yaml -Yaml $raw
    }

    # Lightweight parser fallback: support basic maps/arrays/strings (good enough for CI if needed)
    throw "No YAML parser available. Install PowerShell 7+ or a YAML module (`PowerShellYaml` or `YamlDotNet` wrapper)."
}

function Get-PromptSpec {
    <#
    .SYNOPSIS
      Load a prompt spec (YAML) into a hashtable.
    #>
    [CmdletBinding()] param(
        [Parameter(Mandatory)][string]$Id
    )
    $rel = ($Id -replace '\.','/')
    $path = Join-Path $Script:PromptRoot "catalog/$($rel).yaml"
    if (-not (Test-Path -LiteralPath $path)) { throw "Prompt not found: $($Id) at $($path)" }
    $spec = Resolve-Yaml -Path $path
    $spec.__path = $path
    return $spec
}

function Find-PromptSpec {
    <#
    .SYNOPSIS
      Fuzzy search prompt specs by id/desc/tags.
    #>
    [CmdletBinding()] param(
        [string]$Query = ""
    )
    $files = Get-ChildItem -LiteralPath (Join-Path $Script:PromptRoot 'catalog') -Recurse -Filter *.yaml
    foreach($f in $files){
        $spec = Resolve-Yaml -Path $f.FullName
        $hay  = @($spec.id,$spec.description) + @($spec.tags | ForEach-Object { $_ })
        if ($Query -eq "" -or ($hay -join ' ' -match [regex]::Escape($Query))) {
            [PSCustomObject]@{
                Id          = $spec.id
                Version     = $spec.version
                Description = $spec.description
                Tags        = ($spec.tags -join ',')
                Path        = $f.FullName
            }
        }
    }
}

function Test-PromptSpecSchema {
    <#
    .SYNOPSIS
      Validate a *model output* JSON file against the schema referenced by a prompt spec.
    .DESCRIPTION
      This guards inter-agent contracts. Fail fast if JSON shape drifts.
    #>
    [CmdletBinding()] param(
        [Parameter(Mandatory)][string]$PromptId,
        [Parameter(Mandatory)][string]$JsonPath
    )
    $spec = Get-PromptSpec -Id $PromptId
    $schemaRel = $spec.constraints.schema
    if (-not $schemaRel) { throw "Spec $($PromptId) has no constraints.schema" }
    $schemaPath = if ([Path]::IsPathRooted($schemaRel)) { $schemaRel } else { Join-Path $Script:RepoRoot $schemaRel }
    if (-not (Test-Path -LiteralPath $schemaPath)) { throw "Schema not found: $($schemaPath)" }
    if (-not (Test-Path -LiteralPath $JsonPath)) { throw "JSON not found: $($JsonPath)" }

    $jsonObj = Get-Content -LiteralPath $JsonPath -Raw | ConvertFrom-Json
    $schema  = Get-Content -LiteralPath $schemaPath -Raw | ConvertFrom-Json

    # Minimal draft-07 validator via Newtonsoft.Json.Schema if present, otherwise basic checks
    if (Get-Module -ListAvailable | Where-Object { $_.Name -match 'NJsonSchema|JsonSchema.Net' }) {
        throw "Wire your chosen JSON Schema module here to get full validation."  # leave a clear TODO marker
    }

    # Basic presence check fallback
    foreach($req in $schema.required){
        if ($null -eq $jsonObj.$req) { throw "Schema required property missing: $($req)" }
    }
    return $true
}

function Invoke-PromptTemplate {
    <#
    .SYNOPSIS
      Render a prompt template by applying input variables.
    .DESCRIPTION
      Uses simple {{var}} replacement. For advanced templating, swap a real engine later.
    #>
    [CmdletBinding()]
    param(
        [Parameter(Mandatory)][string]$PromptId,
        [Parameter()][hashtable]$Vars = @{}
    )
    $spec = Get-PromptSpec -Id $PromptId
    $tpl  = [string]$spec.template
    foreach($kv in $Vars.GetEnumerator()){
        $k = [regex]::Escape($kv.Key)
        $v = [string]$kv.Value
        $tpl = [regex]::Replace($tpl, "{{$k}}", [System.Text.RegularExpressions.MatchEvaluator]{ param($m) $v })
    }
    [PSCustomObject]@{
        Id        = $spec.id
        Version   = $spec.version
        Rendered  = $tpl
        ModelHints= $spec.model_hints
    }
}

function Export-PromptIndex {
    <#
    .SYNOPSIS
      Generate prompt catalog indexes (CSV + Excel if ImportExcel is available).
    #>
    [CmdletBinding()] param()
    $rows = Find-PromptSpec | Sort-Object Id
    if (-not $rows) { throw "No prompts found." }
    $csv = Join-Path $Script:IndexRoot "prompt-index.csv"
    $rows | Export-Csv -NoTypeInformation -Encoding UTF8 -Path $csv

    # If ImportExcel is installed, also emit .xlsx (requested preference)
    if (Get-Module -ListAvailable -Name ImportExcel) {
        $xlsx = Join-Path $Script:IndexRoot "prompt-index.xlsx"
        $rows | Export-Excel -Path $xlsx -WorksheetName 'Prompts' -TableName 'PromptIndex' -AutoSize
    }
    return $csv
}

function Test-PromptRepo {
    <#
    .SYNOPSIS
      Lint the entire prompt repository: YAML load, required fields, schema paths exist.
    #>
    [CmdletBinding()] param()
    $errors = @()
    $files = Get-ChildItem -LiteralPath (Join-Path $Script:PromptRoot 'catalog') -Recurse -Filter *.yaml
    foreach($f in $files){
        try{
            $s = Resolve-Yaml -Path $f.FullName
            foreach($req in @('id','role','version','description','template')){ if(-not $s.$req){ throw "Missing '$($req)'" } }
            if($s.constraints -and $s.constraints.schema){
                $schemaPath = if ([Path]::IsPathRooted($s.constraints.schema)) { $s.constraints.schema } else { Join-Path $Script:RepoRoot $s.constraints.schema }
                if(-not (Test-Path -LiteralPath $schemaPath)){ throw "Schema not found: $($schemaPath)" }
            }
        } catch {
            $errors += [PSCustomObject]@{ Path=$f.FullName; Error=$_.Exception.Message }
        }
    }
    if($errors.Count){
        $errors | Format-Table -AutoSize | Out-String | Write-Error
        return $false
    }
    return $true
}

Add a YAML under /prompts/catalog/<domain>/<name>.yaml.

Run Test-PromptRepo to catch missing fields/schema paths.

Render locally with Invoke-PromptTemplate -PromptId <id> -Vars @{...}.

Validate outputs with Test-PromptSpecSchema -PromptId <id> -JsonPath out.json.

Export an index for browsing: Export-PromptIndex (CSV; Excel if ImportExcel is present).
