# Function to export data to JSON
function Export-ToJson {
    param (
        [string]$query,
        [string]$outputFile
    )

    # Execute SQLite command and convert output to JSON
    $json = sqlite3 $databasePath $query | ConvertTo-Json
    Set-Content -Path $outputFile -Value $json
}

# Export People

}
