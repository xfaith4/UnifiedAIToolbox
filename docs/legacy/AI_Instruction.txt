BEGIN: Get-ImageDate-Fixed

function Get-ImageDate {
[CmdletBinding()]
param([Parameter(Mandatory)][string]$Path)

if (-not (Test-Path -LiteralPath $Path)) {
    throw "Path not found: $Path"
}

$exif = & python .\extract_metadata.py $Path 2>$null | ConvertFrom-Json

# Try EXIF first with strict parsing
if ($exif -and $exif.DateTimeOriginal) {
    $s = [string]$exif.DateTimeOriginal
    $dt = $null
    # Common EXIF formats: "yyyy:MM:dd HH:mm:ss" or "yyyy:MM:dd"
    if ([datetime]::TryParseExact($s, @('yyyy:MM:dd HH:mm:ss','yyyy:MM:dd'),
        [System.Globalization.CultureInfo]::InvariantCulture,
        [System.Globalization.DateTimeStyles]::None, [ref]$dt)) {
        return $dt.ToString('yyyy-MM-dd')
    }
}

# Fallback to file mtime
$file = Get-Item -LiteralPath $Path
return $file.LastWriteTime.ToString('yyyy-MM-dd')


}

END: Get-ImageDate-Fixed

File with EXIF “2020:07:05 12:34:56” → “2020-07-05”

File without EXIF → mtime formatted “yyyy-MM-dd”
D) Risks: If EXIF format differs, extend the formats array. Rollback by restoring original function.

Skip rows already processed; parameterize; batch commits; cache EXIF by (Path, LastWriteTime)

Prepared update

$upd.CommandText = "UPDATE FileMetadata SET NewPath=@p WHERE Id=@id;"

$tx = $conn.BeginTransaction()
$upd.Transaction = $tx

$i++

$fi = Get-Item -LiteralPath $row.Path
$key = "$($row.Path)|$($fi.LastWriteTimeUtc.Ticks)"

if (-not $exifCache.ContainsKey($key)) {
    $ex = & python .\extract_metadata.py $row.Path 2>$null | ConvertFrom-Json

    $ex = $exifCache[$key]
}

$dt = $null
if ($ex -and $ex.DateTimeOriginal) {
    [void][datetime]::TryParseExact([string]$ex.DateTimeOriginal, @('yyyy:MM:dd HH:mm:ss','yyyy:MM:dd'),

BEGIN: Trace-Helpers

function Trace { param([string]$Msg,[ConsoleColor]$Color='Cyan'); $ts=Get-Date -Format 'HH:mm:ss'; Write-Host "[$ts] $Msg" -ForegroundColor $Color }
function TraceErr { param([string]$Msg); Trace $Msg 'Red' }

BEGIN: Get-ProfileStorePath

function Get-ProfileStorePath {
[CmdletBinding(SupportsShouldProcess)]
param()
$root = Join-Path $PSScriptRoot ".photo-reorg"
if (-not (Test-Path $root)) {
if ($PSCmdlet.ShouldProcess($root,'Create directory')) {
New-Item -ItemType Directory -Path $root -Force | Out-Null
}
}
Join-Path $root "profiles.json"
}

END: Get-ProfileStorePath
BEGIN: Load-Profiles

function Load-Profiles {
[CmdletBinding()]
param()
$p = Get-ProfileStorePath
if (Test-Path $p) {
try { return (Get-Content $p -Raw | ConvertFrom-Json -ErrorAction Stop) }
catch { TraceErr "[ERROR] profiles.json invalid: $($_.Exception.Message)"; throw }
}
# default
[pscustomobject]@{
current = "default"
profiles = @{ default = @{ DbPath=""; PhotoRoot=""; BaseUrl=""; FilesTable="FileMetadata" } }
}
}

END: Load-Profiles
BEGIN: Save-Profiles

function Save-Profiles {
[CmdletBinding(SupportsShouldProcess)]
param([Parameter(Mandatory)][object]$Object)

$path = Get-ProfileStorePath
$bak  = "$path.bak"

try {
    $json = $Object | ConvertTo-Json -Depth 10 -ErrorAction Stop
    if ($PSCmdlet.ShouldProcess($path,'Write profiles.json with backup')) {
        if (Test-Path $path) { Copy-Item -LiteralPath $path -Destination $bak -Force }
        Set-Content -Path $path -Value $json -Encoding UTF8 -ErrorAction Stop
        Trace "[INFO] Saved profiles to $path"
    }
} catch {
    TraceErr "[ERROR] Save-Profiles failed: $($_.Exception.Message)"
    throw
}


}

END: Save-Profiles
END FILE: ProfilesManager.ps1
END: Build-Template (Example 3, v2)
