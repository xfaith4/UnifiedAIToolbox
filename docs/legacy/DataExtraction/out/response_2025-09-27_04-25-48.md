        window.onload = function() {
            const fields = ['domain', 'format', 'desc', 'focus', 'data'];

                if (saved && document.getElementById(field)) {

                }
            });

            const fields = ['domain', 'format', 'desc', 'focus', 'data'];
            fields.forEach(field => {
                const element = document.getElementById(field);
                if (element) {
                    localStorage.setItem('de_' + field, element.value);
                }
            });
        });

        function clearForm() {
            if (confirm('Clear all fields? This will remove all entered data.')) {
                document.getElementById('analysisForm').reset();
                const fields = ['domain', 'format', 'desc', 'focus', 'data'];
