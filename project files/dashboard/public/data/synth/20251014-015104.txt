# Step 2: Export SQLite data to JSON
function Export-DataToJSON {
    param (
        [string]$databasePath
    )

    # Connect to SQLite database
    $connectionString = "Data Source=$databasePath"
    $connection = New-Object System.Data.SQLite.SQLiteConnection($connectionString)
    $connection.Open()

    # Export People
    $peopleQuery = "SELECT * FROM People"
    $peopleCommand = New-Object System.Data.SQLite.SQLiteCommand($peopleQuery, $connection)
    $peopleAdapter = New-Object System.Data.SQLite.SQLiteDataAdapter($peopleCommand)
    $peopleDataSet = New-Object System.Data.DataSet
    $peopleAdapter.Fill($peopleDataSet)
    $peopleJson = $peopleDataSet.Tables[0] | ConvertTo-Json
    Set-Content -Path "G:\Storage\BenStuff\Development\FamilyTreeModern-5\data\People.json" -Value $peopleJson

    # Export Relationships
    $relationshipsQuery = "SELECT * FROM Relationships"
    $relationshipsCommand = New-Object System.Data.SQLite.SQLiteCommand($relationshipsQuery, $connection)
    $relationshipsAdapter = New-Object System.Data.SQLite.SQLiteDataAdapter($relationshipsCommand)
    $relationshipsDataSet = New-Object System.Data.DataSet
    $relationshipsAdapter.Fill($relationshipsDataSet)
    $relationshipsJson = $relationshipsDataSet.Tables[0] | ConvertTo-Json
    Set-Content -Path "G:\Storage\BenStuff\Development\FamilyTreeModern-5\data\Relationships.json" -Value $relationshipsJson

    # Optional: Export Locations if they exist
    $locationsQuery = "SELECT * FROM Locations"
    $locationsCommand = New-Object System.Data.SQLite.SQLiteCommand($locationsQuery, $connection)
    $locationsAdapter = New-Object System.Data.SQLite.SQLiteDataAdapter($locationsCommand)
    $locationsDataSet = New-Object System.Data.DataSet
    $locationsAdapter.Fill($locationsDataSet)
    if ($locationsDataSet.Tables[0].Rows.Count -gt 0) {
        $locationsJson = $locationsDataSet.Tables[0] | ConvertTo-Json
        Set-Content -Path "G:\Storage\BenStuff\Development\FamilyTreeModern-5\data\Places.json" -Value $locationsJson
    }

    $connection.Close()
}

# Call the function
